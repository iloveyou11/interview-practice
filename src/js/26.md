### IIFE（立即调用表达式）打印结果

```js
var name = 'Tom';
(function() {
    // console.log(name) // undefined
    if (typeof name == 'undefined') {
        var name = 'Jack';
        console.log('Goodbye ' + name);
    } else {
        console.log('Hello ' + name);
    }
})();
// Goodbye Jack
```
